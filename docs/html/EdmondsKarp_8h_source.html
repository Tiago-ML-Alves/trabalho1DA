44 // Mark all vertices as not visited

83template <class T>
84double findMinResidualAlongPath(Vertex<T> *source, Vertex<T> *sink) {
85 double bottleneck = INF;
86 Vertex<T>* cur = sink;
87 while (cur != source)
88 {
89 Edge<T>* e = cur->getPath();
90 if (e == nullptr) break;
91 if (e->getDest() == cur)
92 {
93 bottleneck = std::min(bottleneck, e->getWeight() - e->getFlow());
94 cur = e->getOrig();
95 }
96 else
97 {

99 cur = e->getDest();
100 }
101 }
